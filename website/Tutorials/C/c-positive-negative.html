#include <stdio.h>

/* C program to check whether number is POSITIVE, NEGATIVE or ZERO until user doesn’t want to exit.*/

int main()
{
  int n;
  char choice;
 printf("\n\nWant to check POSITIVE/NEGATIVE Number (press Y/y for 'yes') :");
 fflush(stdin);
 scanf(" %c", &choice);
  while (choice == 'Y' || choice == 'y')
  {
    printf("Enter any number number :");
    scanf("%d", &n);
    if (n > 0)
      printf("\n %d is POSITIVE Number .",n);
    else if (n == 0)
      printf("Number is ZERO.");
    else

    scanf(" %c", &choice);     /*Here is a space before %c*/
  }
printf("\nBye Bye!!!");
return 0;
}

7. C Program to check entered number is ZERO, POSITIVE or NEGATIVE until user does not want to quit. using do while loop.

/* C Program to check entered number is ZERO, POSITIVE or NEGATIVE until user does not want to quit. using do while loop */
#include <stdio.h>

/* C program to check whether number is POSITIVE, NEGATIVE or ZERO until user doesn’t want to exit.*/

int main()
{
  int n;
  char choice;
  do
  {
    printf("Enter any number number :");
    scanf("%d", &n);
    if (n > 0)
      printf("\n %d is POSITIVE Number .",n);
    else if (n == 0)
      printf("Number is ZERO.");
    else
      printf("\n %d is NEGATIVE Number .",n);
    printf("\n\nWant to continue again (press Y/y for 'yes') :");
    fflush(stdin);     /*to clear input buffer*/
    scanf(" %c", &choice);     /*Here is a space before %c*/
  }while (choice == 'Y' || choice == 'y');
printf("\nBye Bye!!!");
return 0;
}

15
15 is POSITIVE Number.

Want to check POSITIVE/NEGATIVE Number (press Y/y for 'yes') :
y
Enter any number number :
-15
-15 is NEGATIVE Number.

Want to check POSITIVE/NEGATIVE Number (press Y/y for 'yes') :
n
Bye Bye!!!
Program Explanation:
Working of this program is same as the given program above(using while loop) except the body execute at least once even though the condition is false.
do while loop is mostly used for menu driven programming.
